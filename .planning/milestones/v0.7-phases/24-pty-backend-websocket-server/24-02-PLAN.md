    **Origin validation (D-11, WS-04 — CSWSH prevention):**
    - ALLOWED_ORIGINS = ["http://localhost:3000", "http://127.0.0.1:3000", "http://localhost:3001"]
    - In the "connection" handler: check `req.headers.origin`
    - If origin is not in the allowlist (or is undefined but NOT coming from a same-host connection):
      - For standalone WSS, the correct approach is to use `wss = new WebSocketServer({ noServer: true })` with an HTTP server that validates in "upgrade" event
      - BUT simpler and correct: use `WebSocketServer({ port: 3001 })` and in "connection" handler check origin — close immediately if invalid
      - Actually implement it with noServer + HTTP server for correct 403 response:

    ```typescript
    import http from "http";
    import { WebSocketServer, WebSocket } from "ws";

    const ALLOWED_ORIGINS = new Set([
      "http://localhost:3000",
      "http://127.0.0.1:3000",
    ]);

    export async function startWsServer(): Promise<void> {
      const httpServer = http.createServer((_req, res) => {
        res.writeHead(426, { "Content-Type": "text/plain" });
        res.end("WebSocket upgrade required");
      });

      const wss = new WebSocketServer({ noServer: true });

      httpServer.on("upgrade", (req, socket, head) => {
        const origin = req.headers.origin ?? "";
        if (!ALLOWED_ORIGINS.has(origin)) {
          console.error(`[ws-server] Rejected origin: ${origin}`);
          socket.write(
            "HTTP/1.1 403 Forbidden\r\nContent-Type: text/plain\r\n\r\nForbidden"
          );
          socket.destroy();
          return;
        }
        wss.handleUpgrade(req, socket, head, (ws) => {
          wss.emit("connection", ws, req);
        });
      });

      wss.on("connection", handleConnection);

      await new Promise<void>((resolve) => {
        httpServer.listen(3001, () => {
          console.error("[ws-server] WebSocket server listening on port 3001");
          resolve();
        });
      });
    }
    ```

    **Connection handler (D-10, D-12, D-13, D-14, D-15, WS-02, WS-03):**

    ```typescript
    const DISCONNECT_TIMEOUT_MS = 30_000; // 30 seconds
    const BATCH_INTERVAL_MS = 8;           // 8ms output batching
    const SEND_BUFFER_MAX = 64 * 1024;    // 64 KB max buffered

    function handleConnection(ws: WebSocket, req: IncomingMessage): void {
      // Parse taskId from URL: /terminal?taskId=xxx
      const url = new URL(req.url ?? "/", "http://localhost:3001");
      const taskId = url.searchParams.get("taskId");

      if (!taskId) {
        console.error("[ws-server] Connection rejected: missing taskId");
        ws.close(1008, "Missing taskId");
        return;
      }

      console.error(`[ws-server] Connection for task ${taskId}`);

      // Check for existing session (reconnect case — WS-03)
      let session = getSession(taskId);

      if (session) {
        // Cancel pending destroy timer
        if (session.disconnectTimer) {
          clearTimeout(session.disconnectTimer);
          session.disconnectTimer = null;
        }
        // Replay ring buffer so client sees what it missed
        const buffer = session.getBuffer();
        if (buffer && ws.readyState === WebSocket.OPEN) {
          ws.send(buffer);
        }
        console.error(`[ws-server] Reconnected to existing session for task ${taskId}`);
      } else {
        // New session — spawn PTY
        // Per v0.7 roadmap: Claude CLI PTY mode uses no --output-format or --print flags
        // Phase 26 will pass the actual command; for now spawn bash as a test shell
        session = createSession(
          taskId,
          "bash",
          [],
          process.cwd(),
          makeBatchedSender(ws),
          (exitCode) => {
            console.error(`[ws-server] PTY exited for task ${taskId} with code ${exitCode}`);
            // Session cleans itself up via onExit (killed=true set in PtySession)
          }
        );
        console.error(`[ws-server] Created new PTY session for task ${taskId}`);
      }

      // WS → PTY input forwarding (D-12, D-13)
      ws.on("message", (rawData) => {
        const data = rawData.toString();
        // D-13: Detect resize message (JSON with type="resize")
        if (data.startsWith("{")) {
          try {
            const msg = JSON.parse(data) as { type?: string; cols?: number; rows?: number };
            if (msg.type === "resize" && typeof msg.cols === "number" && typeof msg.rows === "number") {
              session!.resize(msg.cols, msg.rows);
              return;
            }
          } catch {
            // Not JSON — treat as terminal input
          }
        }
        session!.write(data);
      });

      // D-14: WS disconnect — don't kill PTY, start 30s timer
      ws.on("close", () => {
        console.error(`[ws-server] WS disconnected for task ${taskId} — starting ${DISCONNECT_TIMEOUT_MS}ms keepalive timer`);
        const s = getSession(taskId);
        if (!s || s.killed) return;
        s.disconnectTimer = setTimeout(() => {
          console.error(`[ws-server] Keepalive expired for task ${taskId} — destroying session`);
          destroySession(taskId);
        }, DISCONNECT_TIMEOUT_MS);
      });

      ws.on("error", (err) => {
        console.error(`[ws-server] WS error for task ${taskId}:`, err.message);
      });
    }

    /** D-15: Batch PTY output with 8ms window + bufferedAmount guard */
    function makeBatchedSender(ws: WebSocket): (data: string) => void {
      let pending = "";
      let timer: ReturnType<typeof setTimeout> | null = null;

      return (data: string) => {
        pending += data;
        if (timer) return;
        timer = setTimeout(() => {
          timer = null;
          if (ws.readyState === WebSocket.OPEN && ws.bufferedAmount < SEND_BUFFER_MAX) {
            ws.send(pending);
          }
          pending = "";
        }, BATCH_INTERVAL_MS);
      };
    }
    ```
