    ```typescript
    export function matchGitPathRule(url: string, rules: GitPathRule[]): string {
      if (!rules.length) return "";
      const trimmed = url.trim();
      if (!trimmed) return "";
      const parsed = parseGitUrl(trimmed);
      if (!parsed) return "";

      const { host, pathSegments } = parsed;
      const owner = pathSegments[0] ?? "";
      const repo = pathSegments[pathSegments.length - 1] ?? "";

      const sorted = [...rules].sort((a, b) => a.priority - b.priority);

      for (const rule of sorted) {
        if (rule.host !== host) continue;
        if (rule.ownerMatch !== "*" && rule.ownerMatch !== owner) continue;
        return expandHome(
          rule.localPathTemplate
            .replace("{owner}", owner)
            .replace("{repo}", repo)
        );
      }
      return "";
    }
    ```

    IMPORTANT: Do NOT import anything from `@/actions/config-actions` in this file. The function takes rules as a parameter — it does not fetch them. This keeps git-url.ts free of Next.js/server imports (project constraint from STATE.md).

    IMPORTANT: Use `[...rules].sort()` not `rules.sort()` to avoid mutating the input array.

    2. Add `resolveGitLocalPath` server action at the end of the file, per D-13:
    ```typescript
    export async function resolveGitLocalPath(url: string): Promise<string> {
      const trimmed = url.trim();
      if (!trimmed) return "";
      try {
        const rules = await getConfigValue<GitPathRule[]>("git.pathMappingRules", []);
        const matched = matchGitPathRule(trimmed, rules);
        if (matched) return matched;
        return gitUrlToLocalPath(trimmed);
      } catch {
        return gitUrlToLocalPath(trimmed);
      }
    }
    ```

    The function: reads rules from DB via getConfigValue, tries matchGitPathRule first, falls back to existing hardcoded gitUrlToLocalPath if no rule matches. try/catch ensures the fallback also fires on any DB error.
