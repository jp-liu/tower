```typescript
"use client";

import { ThemeProvider as NextThemesProvider } from "next-themes";
import type { ThemeProviderProps } from "next-themes";

export function ThemeProvider({ children, ...props }: ThemeProviderProps) {
  return <NextThemesProvider {...props}>{children}</NextThemesProvider>;
}
```
This is per D-07 — wrapping next-themes in a "use client" component.

2. Modify `src/app/layout.tsx` per D-07 and D-08:
   - Add import: `import { ThemeProvider } from "@/components/providers/theme-provider";`
   - Add `suppressHydrationWarning` to `<html>` tag: `<html lang="zh-CN" suppressHydrationWarning>`
   - Wrap the provider tree with ThemeProvider OUTSIDE TooltipProvider (per D-08, theme is outside i18n):
   ```
   <ThemeProvider attribute="class" defaultTheme="system" enableSystem>
     <TooltipProvider>
       <I18nProvider>
         <LayoutClient workspaces={workspaces}>
           {children}
         </LayoutClient>
       </I18nProvider>
     </TooltipProvider>
   </ThemeProvider>
   ```
   The key props: `attribute="class"` (matches @custom-variant pattern), `defaultTheme="system"`, `enableSystem` (for GNRL-03).
