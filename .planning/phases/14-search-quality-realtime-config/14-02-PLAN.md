    Current debounced search useEffect (lines 90-100):
    ```typescript
    useEffect(() => {
      if (timerRef.current) clearTimeout(timerRef.current);
      if (!query.trim()) { setResults([]); return; }
      setIsSearching(true);
      timerRef.current = setTimeout(async () => {
        const r = await globalSearch(query, category);
        setResults(r);
        setIsSearching(false);
      }, debounceMs);
      return () => { if (timerRef.current) clearTimeout(timerRef.current); };
    }, [query, category, debounceMs]);
    ```

    Replace with:
    ```typescript
    useEffect(() => {
      if (timerRef.current) clearTimeout(timerRef.current);
      if (!query.trim()) { setResults([]); setIsSearching(false); return; }
      setIsSearching(true);

      let cancelled = false;

      timerRef.current = setTimeout(async () => {
        const r = await globalSearch(query, category);
        if (!cancelled) {
          setResults(r);
          setIsSearching(false);
        }
      }, debounceMs);

      return () => {
        if (timerRef.current) clearTimeout(timerRef.current);
        cancelled = true;
      };
    }, [query, category, debounceMs]);
    ```

    Key details:
    - `let cancelled = false` declared at useEffect body level, OUTSIDE the setTimeout callback (per Pitfall 5 in RESEARCH.md)
    - Both `setResults(r)` and `setIsSearching(false)` are inside the `if (!cancelled)` guard (per D-07 — spinner must not clear on stale completion)
    - The existing `clearTimeout` in cleanup is preserved — it prevents the network call from firing; the cancelled flag prevents state update if the call already started
    - Added `setIsSearching(false)` to the early return for empty query to prevent spinner stuck state

    **Change 2: Config re-fetch on dialog open (D-09)**

    Current code has two separate useEffects (lines 76-87):

    - The separate mount-only `useEffect(() => { ... }, [])` for debounceMs is REMOVED entirely
    - Config fetch moves into the open effect so debounceMs reloads every time the dialog opens

    ```typescript
    describe("SearchDialog - race condition fix (SRCH-07)", () => {
      it("does not display stale results when query changes rapidly", async () => {
        const { globalSearch } = await import("@/actions/search-actions");
        const mockSearch = vi.mocked(globalSearch);

        // First call (stale) resolves slowly with "old" results
        // Second call (current) resolves quickly with "new" results
        let callCount = 0;
        mockSearch.mockImplementation(async (query) => {
          callCount++;
          if (callCount === 1) {
            // Simulate slow response for first query
            await new Promise((r) => setTimeout(r, 500));
            return [{ id: "old", type: "task", title: "Old Result", subtitle: "ws", navigateTo: "/old" }];
          }
          return [{ id: "new", type: "task", title: "New Result", subtitle: "ws", navigateTo: "/new" }];
        });

        renderDialog();

        const input = screen.getByPlaceholderText(/搜索/);
        const userEvent = (await vi.importActual("@testing-library/user-event") as any).default;
        const user = userEvent.setup();

        // Type first query then immediately change to second query
        await user.type(input, "old");
        await user.clear(input);
        await user.type(input, "new");

        // Wait for results to settle
        await vi.waitFor(() => {
          expect(screen.getByText("New Result")).toBeInTheDocument();
        }, { timeout: 2000 });

        // "Old Result" should NOT be present (stale result suppressed by cancelled flag)
        expect(screen.queryByText("Old Result")).not.toBeInTheDocument();
      });
    });
    ```
