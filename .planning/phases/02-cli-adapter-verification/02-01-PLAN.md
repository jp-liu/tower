```typescript
// Check 2: get CLI version (best-effort per D-06)
try {
  const versionProbe = await runChildProcess(
    `claude-version-${Date.now()}`,
    "claude",
    ["--version"],
    {
      cwd,
      env: {},
      timeoutSec: 5,
      graceSec: 2,
      onLog: async () => {},
    },
  );
  const versionStr = (versionProbe.stdout.trim() || versionProbe.stderr.trim()) || "";
  checks.push({
    name: "claude_version",
    passed: true,
    message: versionStr ? `Version: ${versionStr}` : "Version: unknown",
  });
} catch {
  checks.push({
    name: "claude_version",
    passed: true,
    message: "Version: unknown",
  });
}
```

- `passed` is ALWAYS `true` for version check (D-06: never fail the overall test for version)

- Insert AFTER command resolution succeeds but BEFORE API key check
- The `ok` calculation on line 108 already uses `checks.every(c => c.passed || c.name === "anthropic_api_key")` -- version check with `passed: true` does not affect this logic
