**Part A: Update ~/.claude/hooks/notify-agi.sh**

Add the AI_MANAGER_TASK_ID gate immediately after the `log "=== Hook fired ==="` line and stdin reading (after line 30 in current script, after SESSION_ID/CWD/EVENT extraction). Insert:

```bash
# ---- NTFY-03: ai-manager gate ----
# If AI_MANAGER_TASK_ID is set, this is an ai-manager-dispatched session.
# Use structured notification path. If not set, fall through to legacy behavior.
if [ -n "${AI_MANAGER_TASK_ID:-}" ]; then
    log "ai-manager session detected: taskId=$AI_MANAGER_TASK_ID"

    # Read exit code from signal file (written by onExit in agent-actions.ts)
    EXIT_CODE_FILE="/tmp/ai-manager-exit-${AI_MANAGER_TASK_ID}"
    EXIT_CODE=""
    # Retry briefly — onExit may still be writing
    for i in 1 2 3 4 5; do
        if [ -f "$EXIT_CODE_FILE" ]; then
            EXIT_CODE=$(cat "$EXIT_CODE_FILE" 2>/dev/null)
            rm -f "$EXIT_CODE_FILE" 2>/dev/null
            break
        fi
        sleep 0.5
    done

    # Determine status label from exit code
    if [ "$EXIT_CODE" = "0" ]; then
        STATUS="✅ 完成"
    elif [ -n "$EXIT_CODE" ]; then
        STATUS="❌ 失败 (exit $EXIT_CODE)"
    else
        STATUS="⚠️ 未知 (no exit code)"
    fi

    # Compute elapsed time
    TASK_TITLE="${AI_MANAGER_TASK_TITLE:-unknown}"
    STARTED_AT="${AI_MANAGER_STARTED_AT:-}"
    ELAPSED="unknown"
    if [ -n "$STARTED_AT" ]; then
        START_EPOCH=$(date -j -f "%Y-%m-%dT%H:%M:%S" "${STARTED_AT%%.*}" +%s 2>/dev/null || echo "")
        if [ -n "$START_EPOCH" ]; then
            NOW_EPOCH=$(date +%s)
            ELAPSED_S=$(( NOW_EPOCH - START_EPOCH ))
            ELAPSED_M=$(( ELAPSED_S / 60 ))
            ELAPSED_R=$(( ELAPSED_S % 60 ))
            ELAPSED="${ELAPSED_M}m${ELAPSED_R}s"
        fi
    fi

    # Read terminal output for summary (same sources as legacy)
    SUMMARY=""
    TASK_OUTPUT="${RESULT_DIR}/task-output.txt"
    if [ -f "$TASK_OUTPUT" ] && [ -s "$TASK_OUTPUT" ]; then
        SUMMARY=$(tail -c 1000 "$TASK_OUTPUT" | tr '\n' ' ')
    elif [ -f "/tmp/claude-code-output.txt" ] && [ -s "/tmp/claude-code-output.txt" ]; then
        SUMMARY=$(tail -c 1000 /tmp/claude-code-output.txt | tr '\n' ' ')
    fi

    # Determine notification target group
    NOTIFY_GROUP="${FEISHU_NOTIFY_GROUP:-}"
    if [ -z "$NOTIFY_GROUP" ] && [ -f "$META_FILE" ]; then
        NOTIFY_GROUP=$(jq -r '.telegram_group // ""' "$META_FILE" 2>/dev/null || echo "")
    fi

    if [ -n "$NOTIFY_GROUP" ] && [ -x "$OPENCLAW_BIN" ]; then
        MSG="🤖 ai-manager 任务通知
📋 任务: ${TASK_TITLE}
📊 状态: ${STATUS}
⏱️ 耗时: ${ELAPSED}
📝 摘要: ${SUMMARY:0:500}"

        "$OPENCLAW_BIN" message send \
            --channel feishu \
            --account notify \
            --target "$NOTIFY_GROUP" \
            --message "$MSG" 2>/dev/null && log "Sent ai-manager Feishu notification" || log "Feishu send failed"
    else
        log "No FEISHU_NOTIFY_GROUP configured and no meta file group — skipping notification"
    fi

    log "=== ai-manager hook completed ==="
    exit 0
fi

# ---- Legacy behavior (non-ai-manager sessions) continues below ----
```

Place this block AFTER the stdin reading + dedup lock section (after `touch "$LOCK_FILE"` line) but BEFORE the "读取 Claude Code 输出" section. This way, ai-manager sessions exit early with their own notification path, and legacy sessions continue with the existing behavior unchanged.
