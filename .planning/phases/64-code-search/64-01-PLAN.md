    GREEN phase — implement in src/actions/search-code-actions.ts:

    "use server";

    Export types:
    - SearchMatch { filePath: string; lineNumber: number; lineText: string; submatches: Array<{start: number; end: number}> }
    - SearchResult { matches: SearchMatch[]; truncated: boolean; error?: string }

    Export async function searchCode(localPath, pattern, glob?, maxResults=200):
    1. Validate with Zod: localPath (string.min(1)), pattern (string.min(1).max(500)), glob (string.max(200).optional()), maxResults (number.int().min(1).max(500).default(200))
    2. Guard: localPath must start with "/" (absolute path)
    3. Check rg availability: execFileSync("which", ["rg"], { stdio: "pipe" }) — catch → return { matches: [], truncated: false, error: "ripgrep (rg) is not installed. Install with: brew install ripgrep" }
    4. Build args: ["--json", "-n", safePattern] — if glob, push "--glob", safeGlob — push safePath
    5. Call execFileSync("rg", args, { encoding: "utf-8", maxBuffer: 10*1024*1024, stdio: ["ignore","pipe","pipe"] })
    6. Parse output: split by "\n", JSON.parse each line, filter type==="match"
    7. For each match: strip localPath+"/" prefix for filePath, extract line_number, lines.text (trimRight), submatches
    8. Truncate at maxResults, set truncated=true
    9. Return { matches, truncated }
    Catch: if err.status===1, return empty. Else return { matches:[], truncated:false, error: String(err) }

    IMPORTANT:
    - Use vi.hoisted() for child_process mock (required per Phase 62 decision for jsdom vitest)
    - Never use shell:true
    - safeResolvePath NOT needed here — localPath is validated as absolute already; rg is scoped to it directly
  </implementation>
