   function LayoutInner({ workspaces, children, handleCreateProject }: {
     workspaces: LayoutClientProps["workspaces"];
     children: React.ReactNode;
     handleCreateProject: (data: CreateProjectData) => Promise<void>;
   }) {
     const pathname = usePathname();
     const { isOpen, displayMode, closeAssistant } = useAssistant();

     const isTaskDetailPage = /\/workspaces\/[^/]+\/tasks\/[^/]+/.test(pathname);
     const isSubPage = /\/workspaces\/[^/]+\/(notes|assets|archive)/.test(pathname);

     // Sidebar panel — rendered as flex sibling for push layout (UX-02)
     const sidebarPanel = isOpen && displayMode === "sidebar" ? (
       <AssistantPanel mode="sidebar" />
     ) : null;

     // Dialog panel — rendered in Dialog component for modal mode (UI-04)
     const dialogPanel = displayMode === "dialog" ? (
       <Dialog open={isOpen} onOpenChange={(open) => { if (!open) closeAssistant(); }}>
         <DialogContent style={{ maxWidth: "600px", height: "70vh", minHeight: "480px", maxHeight: "800px", padding: 0 }}>
           <AssistantPanel mode="dialog" />
         </DialogContent>
       </Dialog>
     ) : null;

     if (isTaskDetailPage || isSubPage) {
       return (
         <>
           <div className="flex h-screen overflow-hidden">
             {/* Push sidebar for full-screen pages */}
             {sidebarPanel}
             <div className="flex flex-1 flex-col overflow-hidden">
               <TopBar onCreateProject={handleCreateProject} />
               <main className="flex-1 overflow-hidden bg-background">
                 <ErrorBoundary>{children}</ErrorBoundary>
               </main>
             </div>
           </div>
           {dialogPanel}
         </>
       );
     }

     return (
       <>
         <div className="flex h-screen overflow-hidden">
           <AppSidebar workspaces={workspaces} />

               <ErrorBoundary>{children}</ErrorBoundary>
             </main>
           </div>
         </div>
         {dialogPanel}
       </>
     );
   }

**Critical layout detail (UX-02 push sidebar):**
The `AssistantPanel` in sidebar mode is placed as a direct flex sibling inside the `flex h-screen` row. It renders with `w-[420px] shrink-0`, so the main content area (which has `flex-1`) automatically shrinks by 420px. This is the push layout — NOT an overlay. Do NOT wrap the sidebar panel in a Sheet component (Sheet uses `fixed z-50` overlay positioning which violates UX-02).

For the normal layout (with AppSidebar), the sidebar panel is placed AFTER `AppSidebar` and BEFORE the main content `div`. This means the assistant panel appears to the right of the app sidebar and to the left of the main content.

For full-screen pages (isTaskDetailPage || isSubPage), the sidebar panel is the first child in the flex row (there is no AppSidebar on these pages).

**Dialog mode:** The `Dialog` component is rendered outside the main flex container. It only mounts when `displayMode === "dialog"`. The `onOpenChange` handler calls `closeAssistant()` (not just setting state) to ensure the DELETE API call fires and the PTY session is destroyed. The DialogContent has `padding: 0` so the AssistantPanel fills the entire dialog area. Dimensions: 600px wide, 70vh tall (min 480px, max 800px) per UI-SPEC.
